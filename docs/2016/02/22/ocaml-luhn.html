let check_number number =
  let calculate i x =
    if i mod 2 = 0 then
      int_of_string x
    else
      let d = int_of_string x * 2 in d mod 10 + d / 10
  in

  Str.split (Str.regexp "") number
  |> List.rev
  |> List.mapi calculate
  |> List.fold_left (fun x y -> x + y) 0
  |> fun i -> i mod 10 = 0
